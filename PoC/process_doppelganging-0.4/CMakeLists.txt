cmake_minimum_required (VERSION 2.8)
project (proc_doppel)

set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} /MT")

set (srcs
	main.cpp
	util.cpp
	ntdll_undoc.cpp
	pe_hdrs_helper.cpp

)

	util.h
	ntdll_undoc.h
	ntdll_types.h
	ntddk.h
	pe_hdrs_helper.h
	process_env.h
)

add_executable (proc_doppel ${hdrs} ${srcs})
